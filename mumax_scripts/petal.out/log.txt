anisU = vector(0, 0, 1)
a := imageShape("mask.png")
setgeom(a)
m = vortex(-1, 1)
relax()
SaveAs(m, "B=0mT")
for i := 5; i <= 5; i += 2 {
	B_ext = vector(-i*sqrt(2)*1e-3, i*sqrt(2)*1e-3, 0.0)
	minimize()
	SaveAs(m, Sprintf("B=%dmT", i))
}
