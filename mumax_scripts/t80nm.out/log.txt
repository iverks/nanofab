Size := 640
Nt := 2
SetGridsize(Size, Size, Nt)
SetCellsize(5e-9, 5e-9, 40e-9)
Msat = 790e3
Aex = 13e-12
anisU = vector(0, 0, 1)
Ku1 = 10e4
a := cylinder(3000e-9, 80e-9).transl(0, 0, 0)
setgeom(a)
m = vortex(-1, 1)
relax()
SaveAs(m, "B0mT")
for i := 1; i <= 5; i += 2 {
	B_ext = vector(i*1e-3, 0.0, 0.0)
	minimize()
	SaveAs(m, Sprintf("%.0fmT", i))
}
